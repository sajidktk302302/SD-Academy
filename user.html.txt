<script src="https://cdn.jsdelivr.net/npm/bootstrap@5.1.3/dist/js/bootstrap.bundle.min.js"></script>
<script>
    document.addEventListener('DOMContentLoaded', () => {
        let userId = localStorage.getItem('course_app_user_id');
        if (!userId) {
            userId = crypto.randomUUID();
            localStorage.setItem('course_app_user_id', userId);
        }

        const allCourses = {{ courses|tojson }};
        const paymentModal = new bootstrap.Modal(document.getElementById('paymentModal'));

        async function renderCourses() {
            const grid = document.getElementById('course-grid');
            const response = await fetch('/get-status', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ userId: userId })
            });
            const statuses = await response.json();
            
            grid.innerHTML = '';
            allCourses.forEach(course => {
                const status = statuses[course.id] || 'locked';
                let buttonHtml = '';
                
                if (status === 'approved') {
                    buttonHtml = `<a href="${course.download_link}" class="btn btn-success">‚úÖ Download Now</a>`;
                } else if (status === 'pending') {
                    buttonHtml = `<button class="btn btn-warning" disabled>‚è≥ Pending</button>`;
                } else if (status === 'expired') {
                     buttonHtml = `<button class="btn btn-danger" onclick="openPaymentModal(${course.id})">‚è∞ Expired! Unlock Again</button>`;
                } else { // locked or rejected
                    buttonHtml = `<button class="btn btn-primary" onclick="openPaymentModal(${course.id})">üîí Unlock</button>`;
                }

                grid.innerHTML += `
                    <div class="col">
                        <div class="card h-100">
                            <img src="${course.image_url}" class="card-img-top" alt="${course.title}">
                            <div class="card-body">
                                <h5 class="card-title">${course.title}</h5>
                                <p class="card-text">Price: $${course.price}</p>
                                ${buttonHtml}
                            </div>
                        </div>
                    </div>
                `;
            });
        }

        window.openPaymentModal = (courseId) => {
            document.getElementById('course-id-input').value = courseId;
            paymentModal.show();
        };

        document.getElementById('payment-form').addEventListener('submit', async (e) => {
            e.preventDefault();
            const data = {
                userId: userId,
                courseId: document.getElementById('course-id-input').value,
                userName: document.getElementById('user-name').value,
                txId: document.getElementById('tx-id').value,
            };
            
            const response = await fetch('/submit-payment', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(data)
            });
            const result = await response.json();
            alert(result.message);
            if(response.ok) {
                paymentModal.hide();
                renderCourses();
            }
        });

        renderCourses();
    });
</script>
